function createProvincesFromSVG() { // Extract SVG paths from the original iranmap.html fetch('iranmap.html') .then(response => response.text()) .then(html => { const parser = new DOMParser(); const doc = parser.parseFromString(html, 'text/html'); if (shape) { createProvince3D(shape, className, provinceData[className], index); } } }); }) .catch(err => { console.log('Using sample provinces'); // Fallback to sample data Object.keys(provinceData).forEach((key, index) => { createSampleProvince(key, provinceData[key], index); }); }); } function parseSVGPath(pathString) { const shape = new THREE.Shape(); const commands = pathString.match(/[MLHVCSQTAZ][^MLHVCSQTAZ]*/gi); if (!commands) return null; let currentX = 0, currentY = 0; let startX = 0, startY = 0; const scale = 0.5; // Scale down the map try { commands.forEach(cmd => { const type = cmd[0].toUpperCase(); const coords = cmd.slice(1).trim().split(/[\s,]+/).filter(c => c).map(Number); switch (type) { case 'M': currentX = coords[0]; currentY = coords[1]; startX = currentX; startY = currentY; shape.moveTo(currentX * scale, -currentY * scale); break; case 'L': for (let i = 0; i < coords.length; i += 2) { currentX = coords[i]; currentY = coords[i + 1]; shape.lineTo(currentX * scale, -currentY * scale); } break; case 'C': for (let i = 0; i < coords.length; i += 6) { shape.bezierCurveTo( coords[i] * scale, -coords[i + 1] * scale, coords[i + 2] * scale, -coords[i + 3] * scale, coords[i + 4] * scale, -coords[i + 5] * scale ); currentX = coords[i + 4]; currentY = coords[i + 5]; } break; case 'Z': shape.lineTo(startX * scale, -startY * scale); break; } }); return shape; } catch (e) { console.error('Error parsing path:', e); return null; } } function createProvince3D(shape, id, data, index) { const extrudeSettings = { depth: 20, bevelEnabled: true, bevelThickness: 2, bevelSize: 1.5, bevelSegments: 3 }; const geometry = new THREE.ExtrudeGeometry(shape, extrudeSettings);
